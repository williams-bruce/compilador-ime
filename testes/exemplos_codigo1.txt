Exemplo de código:

function fatorial(n : integer) : integer
{
    var i, f: integer;
    i = 2;
    f = 1;

    while( i <= n )
    {
        f = f * i++;
    }

    return f;
}

function somaFat(n : integer) : integer
{
    var i, s: integer;
    i = 0;
    s = 0;

    while( ++i <= n )
    {
        s = s + fatorial(i);
    }

    return s;
}
